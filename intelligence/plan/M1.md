### Algorithmic Logic (to preserve)

**Node descriptions** (`src/intelligence/node-description-generator.ts` line 121):
```typescript
function algorithmicDescription(node: GraphNode, degree: number): string {
  const typeLabel = node.type.charAt(0).toUpperCase() + node.type.slice(1);
  const location = node.source_file ? ` in ${node.source_file}` : "";
  return `${typeLabel} with ${degree} connections${location}.`;
}
```

Selection logic (`node-descriptions.ts` line 159):
```typescript
function selectTargetNodes(nodes: GraphNode[], edgeCounts: Map<string, number>, threshold: number): GraphNode[] {
  const sorted = [...edgeCounts.entries()].sort((a, b) => b[1] - a[1]);
  const cutoff = Math.ceil(sorted.length * (1 - threshold));
  const topNodeIds = new Set(sorted.slice(0, cutoff).map(([id]) => id));
  return nodes.filter((n) => topNodeIds.has(n.id));
}
```

function buildAlgorithmicSummary(nodeCount: number, hubs: GraphNode[], primaryPath: string, repos: string[]): string {

**Algorithmic mode logic** (no provider):

1. Read `graph.json` from `outputDir`
2. Compute edge counts (same as `computeEdgeCounts` in current node-descriptions.ts)
3. Select target nodes by degree threshold (`config.enrich.threshold`)
4. Generate descriptions: `algorithmicDescription(node, degree)`
5. Build community data: group by community, exclude `unclustered` and `<3 nodes`, sort by size, apply `max_communities`
6. For each community: compute hubs, primary path, repos → `buildAlgorithmicSummary()`
7. Copy `graph.json` → `graph-enriched.json` (byte-for-byte via `copyFileSync`)
8. Write `node_descriptions.json` (array of `{id, description}`)
9. Write `community_summaries.json` (array of `{id, label, nodeCount, summary, hub_nodes, primary_path, repos}`)

**Skip logic**:
```
1. Compute sha256(graph.json content) → currentInputHash
2. Compute sha256(JSON.stringify({enabled, threshold, max_communities, provider})) → currentConfigHash
3. If all three output files exist
   AND .cache/enrich-input-hash.txt == currentInputHash
   AND .cache/enrich-config-hash.txt == currentConfigHash
   → SKIP
4. Otherwise → run
5. After completion: write both hash files
```

**Key: extract algorithmic functions** from existing generators into the phase or into a shared module `src/pipeline/enrich/algorithmic.ts`. The generators (`CommunitySummaryGenerator`, `NodeDescriptionGenerator`) can be deleted in Step 4 since they're no longer used.

    expect(existsSync(join(result.outputDir, "community_summaries.json"))).toBe(true);
  });

  it("graph-enriched.json equals graph.json in algorithmic mode", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\ndef call():\n    greet()\n");

    const result = await runBuild(makeConfig(), tmpDir, {});
    const graphJson = readFileSync(join(result.outputDir, "graph.json"), "utf-8");
    const enrichedJson = readFileSync(join(result.outputDir, "graph-enriched.json"), "utf-8");

    expect(enrichedJson).toBe(graphJson);
  });

  it("node_descriptions.json contains entries for high-degree nodes", async () => {
    // Create a project with enough nodes and edges to trigger descriptions
    writeFile(tmpDir, "core.py", [
      "def hub():", "    pass",
      "def caller1():", "    hub()",
      "def caller2():", "    hub()",
      "def caller3():", "    hub()",
      "def caller4():", "    hub()",
    ].join("\n"));

    const config = makeConfig();
    config.enrich.threshold = 0.5;  // top 50%
    const result = await runBuild(config, tmpDir, {});

    const descriptions = JSON.parse(readFileSync(join(result.outputDir, "node_descriptions.json"), "utf-8"));
    expect(Array.isArray(descriptions)).toBe(true);
    // hub node should be described (highest degree)
    if (descriptions.length > 0) {
      expect(descriptions[0]).toHaveProperty("id");
      expect(descriptions[0]).toHaveProperty("description");
      expect(descriptions[0].description).toMatch(/connections/);
    }
  });

  it("community_summaries.json contains summaries for qualifying communities", async () => {
    // Need at least 3 nodes in a community
    writeFile(tmpDir, "auth/service.py", "class AuthService:\n    pass\n");
    writeFile(tmpDir, "auth/token.py", "class TokenManager:\n    pass\n");
    writeFile(tmpDir, "auth/validate.py", "def validate():\n    pass\n");
    writeFile(tmpDir, "api/routes.py", "def route():\n    pass\n");

    const result = await runBuild(makeConfig(), tmpDir, {});

    const summaries = JSON.parse(readFileSync(join(result.outputDir, "community_summaries.json"), "utf-8"));
    expect(Array.isArray(summaries)).toBe(true);
    for (const s of summaries) {
      expect(s).toHaveProperty("id");
      expect(s).toHaveProperty("label");
      expect(s).toHaveProperty("summary");
      expect(s).toHaveProperty("hub_nodes");
      expect(s).toHaveProperty("primary_path");
      expect(s.label).toMatch(/^Community \d+$/);
    }
  });

  it("enrich phase skips on second build (no changes)", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    await runBuild(makeConfig(), tmpDir, {});
    const result2 = await runBuild(makeConfig(), tmpDir, {});

    const enrichResult = result2.phases.get("enrich");
    expect(enrichResult?.skipped).toBe(true);
    expect(enrichResult?.skipReason).toContain("up to date");
  });

  it("enrich phase re-runs when source code changes", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");
    await runBuild(makeConfig(), tmpDir, {});

    // Add a new file → graph changes → enrich re-runs
    writeFile(tmpDir, "extra.py", "def extra():\n    pass\n");
    const result2 = await runBuild(makeConfig(), tmpDir, {});

    const enrichResult = result2.phases.get("enrich");
    expect(enrichResult?.skipped).toBe(false);
  });

  it("enrich.enabled=false produces no output files", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    const config = makeConfig();
    config.enrich.enabled = false;
    const result = await runBuild(config, tmpDir, {});

    expect(existsSync(join(result.outputDir, "graph-enriched.json"))).toBe(false);
    expect(existsSync(join(result.outputDir, "node_descriptions.json"))).toBe(false);
    expect(existsSync(join(result.outputDir, "community_summaries.json"))).toBe(false);
  });

  it("--target enrich stops before Level 4 phases", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    const result = await runBuild(makeConfig(), tmpDir, { target: "enrich" });

    expect(result.phases.has("enrich")).toBe(true);
    expect(result.phases.has("index")).toBe(false);
    expect(result.phases.has("embeddings")).toBe(false);
    expect(result.phases.has("html")).toBe(false);
    expect(result.phases.has("report")).toBe(false);
  });

  it("search-index reads graph-enriched.json (not graph.json)", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    const config = makeConfig();
    config.html = false;
    config.embeddings.enabled = false;
    const result = await runBuild(config, tmpDir, {});

    // index phase ran successfully (no error) and graph_search.db exists
    expect(existsSync(join(result.outputDir, "graph_search.db"))).toBe(true);
    const indexResult = result.phases.get("index");
    expect(indexResult?.skipped).toBe(false);
  });

  it("embeddings phase works with enriched outputs (TF-IDF mode)", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    '''Say hello'''\n    pass\n");
    writeFile(tmpDir, "utils.py", "def helper():\n    '''Help'''\n    pass\n");

    const config = makeConfig();
    config.embeddings.enabled = true;
    // No provider → TF-IDF algorithmic embeddings
    const result = await runBuild(config, tmpDir, {});

    expect(existsSync(join(result.outputDir, "vectors"))).toBe(true);
    const embResult = result.phases.get("embeddings");
    expect(embResult?.skipped).toBe(false);
  });

  it("report phase produces report.md reading graph-enriched.json", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    const config = makeConfig();
    config.html = false;
    config.embeddings.enabled = false;
    const result = await runBuild(config, tmpDir, {});

    expect(existsSync(join(result.outputDir, "report.md"))).toBe(true);
    const report = readFileSync(join(result.outputDir, "report.md"), "utf-8");
    expect(report).toContain("Graph Architecture Report");
  });

  it("empty repo still produces valid (empty) output files", async () => {
    // No source files
    const result = await runBuild(makeConfig(), tmpDir, {});

    const enriched = JSON.parse(readFileSync(join(result.outputDir, "graph-enriched.json"), "utf-8"));
    expect(enriched.nodes).toEqual([]);

    const descriptions = JSON.parse(readFileSync(join(result.outputDir, "node_descriptions.json"), "utf-8"));
    expect(descriptions).toEqual([]);

    const summaries = JSON.parse(readFileSync(join(result.outputDir, "community_summaries.json"), "utf-8"));
    expect(summaries).toEqual([]);
  });

  it("force build re-runs enrich even when cache is valid", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    pass\n");

    await runBuild(makeConfig(), tmpDir, {});
    const result2 = await runBuild(makeConfig(), tmpDir, { force: true });

    const enrichResult = result2.phases.get("enrich");
    expect(enrichResult?.skipped).toBe(false);
  });

  it("build with all phases enabled completes without error", async () => {
    writeFile(tmpDir, "main.py", "def greet():\n    '''Hello'''\n    pass\n");
    writeFile(tmpDir, "utils.py", "def helper():\n    greet()\n");

    const config = makeConfig();
    config.html = true;
    config.embeddings.enabled = true;
    config.outlines.enabled = true;
    const result = await runBuild(config, tmpDir, {});

    // All 8 phases ran
    expect(result.phases.size).toBe(8);
    // No phase failed
    for (const [id, phaseResult] of result.phases) {
      expect(phaseResult.skipReason).not.toMatch(/^error:/);
    }
  });
});
